## 一、测试目标

验证关键 API 接口在 **100 并发**、**30 秒持续负载**下的性能基线：

| 指标 | 目标值 |
|------|--------|
| QPS | ≥ 500 |
| P95 响应时间 | < 200ms |
| 错误率 | < 0.5% |

| 内存使用 | < 70% |

### 4.1 安装 k6

### 4.2 综合压测脚本
```javascript
// load-test.js
import http from 'k6/http';
import { check, sleep } from 'k6';

export const options = {
  stages: [
    { duration: '10s', target: 50 },   // 预热
    { duration: '30s', target: 100 },  // 持续负载
    { duration: '10s', target: 200 },  // 峰值
    { duration: '10s', target: 0 },    // 退出
  ],
  thresholds: {
    http_req_duration: ['p(95)<200'],   // P95 < 200ms
    http_req_failed: ['rate<0.005'],    // 错误率 < 0.5%
  },
};

const BASE_URL = __ENV.BASE_URL || 'http://localhost:3000';

export default function () {
  // 1. 登录
  const loginRes = http.post(
    `${BASE_URL}/api/auth/login`,
    JSON.stringify({ phone: '13800000001', password: 'test123456' }),
    { headers: { 'Content-Type': 'application/json' } }
  );
  check(loginRes, {
    'login status 200': (r) => r.status === 200,
  });
  const token = loginRes.json('data.token');

  const authHeaders = {
    headers: {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${token}`,
    },
  };

  // 2. 技能列表
  const skillsRes = http.get(`${BASE_URL}/api/skills?page=1&pageSize=20`, authHeaders);
  check(skillsRes, {
    'skills status 200': (r) => r.status === 200,
    'skills response < 200ms': (r) => r.timings.duration < 200,
  });

  // 3. 技能详情
  const skillId = skillsRes.json('data.list.0.id');
  if (skillId) {
    const detailRes = http.get(`${BASE_URL}/api/skills/${skillId}`, authHeaders);
    check(detailRes, { 'detail status 200': (r) => r.status === 200 });
  }

  // 4. 厨房列表
  const kitchenRes = http.get(`${BASE_URL}/api/kitchen/posts?page=1&pageSize=20`, authHeaders);
  check(kitchenRes, { 'kitchen status 200': (r) => r.status === 200 });

  // 5. 时间银行
  const tbRes = http.get(`${BASE_URL}/api/time-bank/services?page=1&pageSize=20`, authHeaders);
  check(tbRes, { 'time-bank status 200': (r) => r.status === 200 });

  // 6. 应急
  const emRes = http.get(`${BASE_URL}/api/emergency/requests?page=1&pageSize=20`, authHeaders);
  check(emRes, { 'emergency status 200': (r) => r.status === 200 });

  sleep(1);
}
```

### 4.3 运行压测
```bash
k6 run load-test.js
# 或指定目标地址
k6 run --out json=result.json load-test.js
BASE_URL=https://staging.example.com k6 run load-test.js
```

## 七、慢查询监控

通过 PostgreSQL 慢查询日志定位瓶颈：

```sql
-- 启用慢查询日志（重启后生效）
ALTER SYSTEM SET log_min_duration_statement = '500ms';

-- 查看最近的慢查询
SELECT
  query,
  calls,
  mean_exec_time,
  total_exec_time
FROM pg_stat_statements
ORDER BY mean_exec_time DESC
LIMIT 20;
```
